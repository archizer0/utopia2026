import React, { useState, useEffect, useRef } from 'react';
import { Camera, SwitchCamera, Download, Sparkles, X, Image as ImageIcon } from 'lucide-react';

export default function App() {
  const videoRef = useRef(null);
  const canvasRef = useRef(null);
  const animationRef = useRef(null);
 
  const [hasPermission, setHasPermission] = useState(false);
  const [error, setError] = useState(null);
  const [facingMode, setFacingMode] = useState('user');
  const [activeFilter, setActiveFilter] = useState('normal');
  const [capturedImage, setCapturedImage] = useState(null);
  const [flash, setFlash] = useState(false);

  // Estado das partículas (Cartas)
  const cardsRef = useRef(Array.from({ length: 15 }, () => ({
    x: Math.random(),
    y: Math.random() * -1,
    speed: 0.005 + Math.random() * 0.015,
    rotation: Math.random() * Math.PI * 2,
    rotSpeed: (Math.random() - 0.5) * 0.1,
    suit: ['♥️', '♦️', '♠️', '♣️'][Math.floor(Math.random() * 4)],
    color: Math.random() > 0.5 ? '#e11d48' : '#1e293b'
  })));

  // Filtros disponíveis
  const filters = [
    { id: 'normal', name: 'Mundo Real', icon: '🌍' },
    { id: 'rabbitHole', name: 'Buraco do Coelho', icon: '🌀' },
    { id: 'shrink', name: 'Poção Beba-me', icon: '🧪' },
    { id: 'madHatter', name: 'Chapeleiro', icon: '🎩' },
    { id: 'cheshire', name: 'Gato de Cheshire', icon: '😸' },
    { id: 'cards', name: 'Rainha de Copas', icon: '🃏' }
  ];

  // Iniciar câmera
  useEffect(() => {
    let stream = null;

    const startCamera = async () => {
      try {
        if (stream) {
          stream.getTracks().forEach(track => track.stop());
        }
       
        stream = await navigator.mediaDevices.getUserMedia({
          video: { facingMode: facingMode, width: { ideal: 1280 }, height: { ideal: 720 } },
          audio: false
        });
       
        if (videoRef.current) {
          videoRef.current.srcObject = stream;
          setHasPermission(true);
        }
      } catch (err) {
        console.error("Erro ao acessar a câmera:", err);
        setError("Não foi possível acessar a câmera. Verifique as permissões.");
      }
    };

    startCamera();

    return () => {
      if (stream) {
        stream.getTracks().forEach(track => track.stop());
      }
      if (animationRef.current) {
        cancelAnimationFrame(animationRef.current);
      }
    };
  }, [facingMode]);

  // Loop de renderização do Canvas
  useEffect(() => {
    if (!hasPermission) return;

    const video = videoRef.current;
    const canvas = canvasRef.current;
    if (!video || !canvas) return;

    const ctx = canvas.getContext('2d');

    const drawCover = (ctx, video, canvasWidth, canvasHeight) => {
      if (video.videoWidth === 0) return; // Vídeo ainda não carregou
     
      const videoRatio = video.videoWidth / video.videoHeight;
      const canvasRatio = canvasWidth / canvasHeight;
      let drawWidth, drawHeight, offsetX, offsetY;

      if (videoRatio > canvasRatio) {
        drawHeight = canvasHeight;
        drawWidth = canvasHeight * videoRatio;
        offsetX = (canvasWidth - drawWidth) / 2;
        offsetY = 0;
      } else {
        drawWidth = canvasWidth;
        drawHeight = canvasWidth / videoRatio;
        offsetX = 0;
        offsetY = (canvasHeight - drawHeight) / 2;
      }

      // Espelhar a imagem se for a câmera frontal
      ctx.save();
      if (facingMode === 'user') {
        ctx.translate(canvasWidth, 0);
        ctx.scale(-1, 1);
      }
      ctx.drawImage(video, offsetX, offsetY, drawWidth, drawHeight);
      ctx.restore();
    };

    const renderLoop = () => {
      // Ajustar tamanho do canvas para corresponder ao contêiner
      const parent = canvas.parentElement;
      if (parent && (canvas.width !== parent.clientWidth || canvas.height !== parent.clientHeight)) {
        canvas.width = parent.clientWidth;
        canvas.height = parent.clientHeight;
      }

      const w = canvas.width;
      const h = canvas.height;
      const time = Date.now() / 1000;

      ctx.clearRect(0, 0, w, h);

      // --- APLICAR FILTROS ---

      if (activeFilter === 'rabbitHole') {
        ctx.save();
        ctx.translate(w/2, h/2);
        ctx.rotate(Math.sin(time) * 0.15); // Rotação psicodélica
        const scale = 1 + Math.sin(time * 2) * 0.15;
        ctx.scale(scale, scale);
        ctx.translate(-w/2, -h/2);
        ctx.filter = `hue-rotate(${time * 50}deg) contrast(1.3) saturate(1.8)`;
        drawCover(ctx, video, w, h);
        ctx.restore();
      }
      else if (activeFilter === 'shrink') {
        // Fundo mágico
        const gradient = ctx.createRadialGradient(w/2, h/2, 0, w/2, h/2, w);
        gradient.addColorStop(0, '#4c1d95');
        gradient.addColorStop(1, '#0f172a');
        ctx.fillStyle = gradient;
        ctx.fillRect(0, 0, w, h);

        // Encolher o vídeo
        const shrinkScale = 0.35 + (Math.sin(time * 1.5) * 0.05); // Pulsação leve
        ctx.save();
        ctx.translate(w/2, h/2);
        ctx.scale(shrinkScale, shrinkScale);
        ctx.translate(-w/2, -h/2);
       
        // Borda ao redor do vídeo
        ctx.shadowColor = 'rgba(255, 255, 255, 0.5)';
        ctx.shadowBlur = 30;
        ctx.lineWidth = 10;
        ctx.strokeStyle = '#fcd34d';
       
        // Desenha o vídeo pequeno
        ctx.save();
        if (facingMode === 'user') {
           ctx.translate(w, 0);
           ctx.scale(-1, 1);
        }
        ctx.drawImage(video, 0, 0, w, h);
        ctx.restore();
       
        ctx.strokeRect(0, 0, w, h);
        ctx.restore();

        // Texto "Beba-me"
        ctx.font = 'bold 36px serif';
        ctx.fillStyle = '#fde047';
        ctx.textAlign = 'center';
        ctx.shadowColor = 'black';
        ctx.shadowBlur = 5;
        ctx.fillText('🍾 Beba-me!', w/2, h - 50);
      }
      else {
        // Desenho normal para os outros filtros
        ctx.filter = 'none';
        drawCover(ctx, video, w, h);
      }

      // --- OVERLAYS SOBRE O VÍDEO ---
     
      if (activeFilter === 'madHatter') {
        // Cartola no topo
        ctx.font = '120px Arial';
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
       
        const yOffset = Math.sin(time * 3) * 10; // Levita um pouco
        ctx.fillText('🎩', w / 2, h * 0.2 + yOffset);
       
        // Etiqueta 10/6
        ctx.save();
        ctx.translate(w / 2 + 30, h * 0.2 + yOffset + 20);
        ctx.rotate(0.2);
        ctx.fillStyle = '#fef08a';
        ctx.shadowColor = 'rgba(0,0,0,0.5)';
        ctx.shadowBlur = 4;
        ctx.fillRect(-25, -15, 50, 30);
        ctx.fillStyle = '#000';
        ctx.font = 'bold 16px Courier';
        ctx.fillText('10/6', 0, 0);
        ctx.restore();
      }

      if (activeFilter === 'cheshire') {
        // Sorriso fantasmagórico
        const alpha = Math.max(0.1, Math.abs(Math.sin(time * 0.8)));
        ctx.globalAlpha = alpha;
        ctx.font = `${Math.min(w * 0.4, 200)}px Arial`;
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.shadowColor = '#c084fc';
        ctx.shadowBlur = 40;
        ctx.fillText('😸', w / 2, h * 0.7);
        ctx.globalAlpha = 1.0;
        ctx.shadowBlur = 0;
      }

      if (activeFilter === 'cards') {
        // Animação de cartas caindo
        cardsRef.current.forEach(card => {
          card.y += card.speed;
          card.rotation += card.rotSpeed;
          if (card.y > 1.2) card.y = -0.2; // Volta para o topo

          ctx.save();
          ctx.translate(card.x * w, card.y * h);
          ctx.rotate(card.rotation);
         
          // Fundo da carta
          ctx.fillStyle = '#ffffff';
          ctx.shadowColor = 'rgba(0,0,0,0.3)';
          ctx.shadowBlur = 10;
          ctx.fillRect(-20, -30, 40, 60);
         
          // Borda
          ctx.strokeStyle = '#cbd5e1';
          ctx.lineWidth = 2;
          ctx.strokeRect(-20, -30, 40, 60);
         
          // Naipe
          ctx.fillStyle = card.color;
          ctx.font = '24px Arial';
          ctx.textAlign = 'center';
          ctx.textBaseline = 'middle';
          ctx.shadowBlur = 0;
          ctx.fillText(card.suit, 0, 0);
         
          ctx.restore();
        });
      }

      animationRef.current = requestAnimationFrame(renderLoop);
    };

    video.onloadedmetadata = () => {
      video.play();
      renderLoop();
    };

    // Caso o vídeo já tenha carregado os metadados (hot reload)
    if (video.readyState >= 1) {
      renderLoop();
    }

  }, [hasPermission, activeFilter, facingMode]);

  const toggleCamera = () => {
    setFacingMode(prev => prev === 'user' ? 'environment' : 'user');
  };

  const takePhoto = () => {
    if (!canvasRef.current) return;
   
    // Efeito de flash
    setFlash(true);
    setTimeout(() => setFlash(false), 150);

    // Capturar frame
    const dataUrl = canvasRef.current.toDataURL('image/jpeg', 0.9);
    setCapturedImage(dataUrl);
  };

  const downloadPhoto = () => {
    if (!capturedImage) return;
    const link = document.createElement('a');
    link.download = `pais-das-maravilhas-${new Date().getTime()}.jpg`;
    link.href = capturedImage;
    link.click();
  };

  return (
    <div className="min-h-screen bg-slate-950 text-slate-100 font-sans flex flex-col items-center justify-center p-4 overflow-hidden relative">
     
      {/* Elementos Decorativos de Fundo */}
      <div className="absolute top-0 left-0 w-full h-full overflow-hidden pointer-events-none opacity-20">
        <div className="absolute top-10 left-10 text-6xl">🍄</div>
        <div className="absolute bottom-20 right-10 text-6xl animate-pulse">⏱️</div>
        <div className="absolute top-40 right-20 text-4xl transform rotate-45">☕</div>
      </div>

      <div className="w-full max-w-md z-10 flex flex-col gap-6">
       
        {/* Cabeçalho */}
        <div className="text-center space-y-2">
          <h1 className="text-3xl font-serif font-bold text-transparent bg-clip-text bg-gradient-to-r from-purple-400 to-pink-400">
            Câmera das Maravilhas
          </h1>
          <p className="text-slate-400 text-sm">Entre na toca do coelho...</p>
        </div>

        {/* Container da Câmera */}
        <div className="relative aspect-[3/4] sm:aspect-[4/3] bg-slate-900 rounded-3xl overflow-hidden shadow-2xl shadow-purple-900/20 border-4 border-slate-800">
         
          {error ? (
            <div className="absolute inset-0 flex flex-col items-center justify-center p-6 text-center text-red-400">
              <X size={48} className="mb-4" />
              <p>{error}</p>
            </div>
          ) : !hasPermission ? (
            <div className="absolute inset-0 flex flex-col items-center justify-center p-6 text-center">
              <Sparkles size={48} className="mb-4 text-purple-400 animate-pulse" />
              <p>Permita o acesso à câmera para começar a magia.</p>
            </div>
          ) : (
            <>
              {/* Vídeo oculto usado como fonte para o Canvas */}
              <video
                ref={videoRef}
                playsInline
                autoPlay
                muted
                className="hidden"
              />
             
              {/* Canvas onde a mágica acontece */}
              <canvas
                ref={canvasRef}
                className="w-full h-full object-cover"
              />

              {/* Flash Screen */}
              {flash && (
                <div className="absolute inset-0 bg-white z-50 animate-ping opacity-75"></div>
              )}
            </>
          )}

          {/* Modal de Imagem Capturada */}
          {capturedImage && (
            <div className="absolute inset-0 z-40 bg-slate-950/90 backdrop-blur-sm flex flex-col items-center justify-center p-4 animate-in fade-in zoom-in duration-300">
              <img
                src={capturedImage}
                alt="Captura"
                className="w-full max-h-[70%] object-contain rounded-xl border-2 border-slate-700 mb-6 shadow-2xl"
              />
              <div className="flex gap-4">
                <button
                  onClick={downloadPhoto}
                  className="flex items-center gap-2 bg-gradient-to-r from-purple-600 to-pink-600 px-6 py-3 rounded-full font-bold shadow-lg hover:scale-105 transition-transform"
                >
                  <Download size={20} />
                  Salvar
                </button>
                <button
                  onClick={() => setCapturedImage(null)}
                  className="flex items-center gap-2 bg-slate-800 hover:bg-slate-700 px-6 py-3 rounded-full font-bold transition-colors"
                >
                  <X size={20} />
                  Descartar
                </button>
              </div>
            </div>
          )}
        </div>

        {/* Controles Principais */}
        <div className="flex justify-between items-center px-4">
          <button
            onClick={toggleCamera}
            disabled={!hasPermission}
            className="p-4 bg-slate-800 rounded-full hover:bg-slate-700 disabled:opacity-50 transition-colors shadow-lg"
            title="Inverter Câmera"
          >
            <SwitchCamera size={24} />
          </button>

          <button
            onClick={takePhoto}
            disabled={!hasPermission || capturedImage !== null}
            className="p-5 bg-gradient-to-tr from-pink-500 to-purple-600 rounded-full hover:scale-105 active:scale-95 disabled:opacity-50 disabled:hover:scale-100 transition-all shadow-[0_0_30px_rgba(216,180,254,0.4)]"
            title="Tirar Foto"
          >
            <Camera size={36} className="text-white" />
          </button>

          <button
            className="p-4 bg-slate-800 rounded-full hover:bg-slate-700 disabled:opacity-50 transition-colors shadow-lg"
            title="Galeria (Demonstração)"
          >
            <ImageIcon size={24} />
          </button>
        </div>

        {/* Seletor de Filtros */}
        <div className="bg-slate-900 p-4 rounded-3xl border border-slate-800 shadow-xl overflow-x-auto hide-scrollbar">
          <div className="flex gap-4 min-w-max px-2">
            {filters.map((filter) => (
              <button
                key={filter.id}
                onClick={() => setActiveFilter(filter.id)}
                className={`flex flex-col items-center gap-2 p-2 rounded-2xl transition-all ${
                  activeFilter === filter.id
                    ? 'bg-purple-900/50 scale-110 shadow-lg'
                    : 'hover:bg-slate-800 opacity-70 hover:opacity-100'
                }`}
              >
                <div className="text-3xl filter drop-shadow-md">
                  {filter.icon}
                </div>
                <span className="text-[10px] uppercase font-bold tracking-wider w-20 text-center leading-tight">
                  {filter.name}
                </span>
              </button>
            ))}
          </div>
        </div>

      </div>
     
      {/* Estilos customizados para esconder a barra de rolagem dos filtros */}
      <style dangerouslySetInnerHTML={{__html: `
        .hide-scrollbar::-webkit-scrollbar {
          display: none;
        }
        .hide-scrollbar {
          -ms-overflow-style: none;
          scrollbar-width: none;
        }
      `}} />
    </div>
  );
}
